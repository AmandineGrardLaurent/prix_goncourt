CREATE TABLE IF NOT EXISTS selection(
       id_selection INT PRIMARY KEY AUTO_INCREMENT NOT NULL,
       selection_date DATE,
       selection_nb INT
    ) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

    CREATE TABLE IF NOT EXISTS editor(
       id_editor INT PRIMARY KEY AUTO_INCREMENT NOT NULL,
       name VARCHAR(50)
    ) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

    CREATE TABLE IF NOT EXISTS person(
       id_person INT PRIMARY KEY AUTO_INCREMENT NOT NULL,
       last_name VARCHAR(50),
       first_name VARCHAR(50)
    ) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

    CREATE TABLE IF NOT EXISTS author(
       id_author INT PRIMARY KEY AUTO_INCREMENT NOT NULL,
       biography TEXT,
       id_person INT,
       FOREIGN KEY(id_person) 
       REFERENCES person(id_person)
    ) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

    CREATE TABLE IF NOT EXISTS academy_member(
       id_academy_member INT PRIMARY KEY AUTO_INCREMENT NOT NULL,
       is_president BOOL,
       id_person INT,
       FOREIGN KEY(id_person) 
       REFERENCES person(id_person)
    ) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

    CREATE TABLE IF NOT EXISTS book(
       id_book INT PRIMARY KEY AUTO_INCREMENT NOT NULL,
       title VARCHAR(50),
       description TEXT,
       publication_date DATE,
       pages_nb INT,
       ISBN VARCHAR(50),
       price DECIMAL(4,2),
       id_editor INT NOT NULL,
       id_author INT NOT NULL,
       FOREIGN KEY(id_author) REFERENCES author(id_author),
       FOREIGN KEY(id_editor) REFERENCES editor(id_editor)
    ) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

    CREATE TABLE IF NOT EXISTS main_character(
       id_main_character INT PRIMARY KEY AUTO_INCREMENT NOT NULL,
       name VARCHAR(50),
       id_book INT NOT NULL,
       FOREIGN KEY(id_book) REFERENCES book(id_book)
    ) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

    CREATE TABLE IF NOT EXISTS book_selection(
       id_book_selection INT PRIMARY KEY AUTO_INCREMENT NOT NULL,
       id_selection INT,
	   id_book INT,
       FOREIGN KEY(id_book) REFERENCES book(id_book),
       FOREIGN KEY(id_selection) REFERENCES selection(id_selection)
    ) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;
